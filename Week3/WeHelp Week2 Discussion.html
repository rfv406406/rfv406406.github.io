找到 Bubble Sort 經典的演算法程式。
Python
Copy
# O(n^2)， for loop 兩層 都是 c * n 次 def bubble_sort(list_input): # n 個數值須執行 n - 1 次 (將大值推到定位 n - 1 次) for i in range(len(list_input) - 2): # 從頭開始比較至上次比較極限前 for j in range(len(list_input) - i - 1): # 若 j + 1 值較大，則下輪由此值接續比較 if list_input[j] <= list_input[j + 1]: continue # 若 j + 1 值較小，則下輪由 j 值比較( swap(j, j+1) ) else: list_input[j], list_input[j + 1] = list_input[j + 1], list_input[j] return list_input print( bubble_sort( [5,11,6,1,1,8,6,4,6,2,3,2,4,7] ) )

找到 Merge Sort 經典的演算法程式。
Python
Copy
def merge(left, right): res = [] # 只要其中一個 array 還有元素 while len(left) and len(right): # 比較左右遞一個元素，小者取出加入 res if left[0] < right[0]: res.append(left.pop(0)) else: res.append(right.pop(0)) # 若左式仍有元素，代表剩餘元素皆大於 res 內的元素(且兩邊都有排序過了)，將 left 加至 res 尾部 res = res + left if len(left) else res + right return res # O(nlogn)， 須執行 ( logn 次 切分 ) * ( n 次 排序 + 合併(每一輪都須比較 n 次) ) def merge_sort(list_input): # 元素少於 2，不切 if len(list_input) < 2: return list_input # 切成 2 個 array mid = len(list_input) // 2 list_left = list_input[:mid] list_right = list_input[mid:] # 先切到底 (2 個元素一個 array)，再合併 return merge( merge_sort(list_left), merge_sort(list_right)) print( merge_sort( [5,11,6,1,1,8,6,4,6,2,3,2,4,7] ) ) # [1, 1, 2, 2, 3, 4, 4, 5, 6, 6, 6, 7, 8, 11]

function bubble_sort(list_input){ for (let i = 0; i < list_input.length - 2; i++){ for (let j = 0; j < list_input.length - i - 1; j++){ if (list_input[j] <= list_input[j + 1]){ continue; }else{ [list_input[j], list_input[j + 1]] = [list_input[j + 1], list_input[j]]; }; }; }; return list_input; }; function merge(l, r){ let temp = []; let res = []; while (l.length > 0 && r.length > 0){ if (l[0] < r[0]){ temp.push(l.shift()); }else{ temp.push(r.shift()); }; }; if (l.length > 0){ res = [].concat(temp, l); }else{ res = [].concat(temp, r); }; return res; }; function merge_sort(array_input){ if (array_input.length < 2){ return array_input; }; let m = array_input.length / 2; let array_left = array_input.slice(0, m); let array_right = array_input.slice(m); return merge( merge_sort(array_left), merge_sort(array_right)); } function getRandom(x){ return Math.floor(Math.random()*x)+1; }; function getarray(num){ let res = []; for (let i = 0; i < num; i++){ res.push(getRandom(10)); } return res; }; arr = getarray(100) console.time("bubble_sort"); bubble_sort(arr); console.timeEnd("bubble_sort"); arr = getarray(100) console.time("merge_sort"); merge_sort(arr); console.timeEnd("merge_sort"); arr = getarray(1000000) console.time("bubble_sort"); bubble_sort(arr); console.timeEnd("bubble_sort"); arr = getarray(1000000) console.time("merge_sort"); merge_sort(arr); console.timeEnd("merge_sort"); // 100 筆資料 bubble_sort: 7.809ms // 100 筆資料 merge_sort: 0.433ms // 1,000,000 筆資料 bubble_sort: 46:03.046 (m:ss.mmm) // 1,000,000 筆資料 merge_sort: 4:01.268 (m:ss.mmm)
